

\mbox{\hyperlink{Flow}{Flow Control }}

Flow control is about conditions.\hypertarget{Flow_Conditions}{}\doxysection{Conditions}\label{Flow_Conditions}
Condtions amount to Bool boolean states. E.\+G.\+:
\begin{DoxyItemize}
\item x $>$ y
\item i == 12
\item 2 + 2 == 4
\item fruit == \char`\"{}apple\char`\"{}
\item etc ~~~~(assuming etc is a boolean variable...)
\end{DoxyItemize}

Therfore the usual boolean rules apply here.\hypertarget{Flow_Conditionals}{}\doxysubsection{Conditionals}\label{Flow_Conditionals}
\mbox{\hyperlink{}{If}}


\begin{DoxyVerbInclude}
# if Conditional Example

fun main():no_value
{
    if true:
    {
        writeln(8)
    }
    
    // ternary expression
    var result:string
    result ? "Yes." : "No!"
    writeln(logical)
}
\end{DoxyVerbInclude}


\mbox{\hyperlink{}{Else}}


\begin{DoxyVerbInclude}
# if - else Conditional Example

fun main():no_value
{
    if true:
    {
        writeln(8)
    }
    else:
    {
        writeln(2 + 3)
    }
}
\end{DoxyVerbInclude}


\mbox{\hyperlink{}{Elsif}}


\begin{DoxyVerbInclude}
# Full Conditional Example

fun main():no_value
{
    if true:
    {
        writeln(8)
    }
    elsif false:
    {
        writeln(7, 9)
    }
    else:
    {
        writeln(2 + 3)
    }
}
\end{DoxyVerbInclude}

\begin{DoxyVerbInclude}

# while loop syntax sample

fun main(argsv):int8
{
    while true:
    {
        # This is the loop that never ends.
    }
    
    while 1 > 3:
    {
        # This loop is skipped.
    }
    
    var i:int8 = 0
    while i < argsv[1]:
    {
        writeln(i)
        i ++
    }

\mbox{\hyperlink{}{For}}


\begin{DoxyVerbInclude}
# forloop.bill
# for loop syntax sample

/*  @fn     main
 *  @brief  forloops
 */
fun main():no_value
{
    for var i:int64 = 0 to 10:
    {
        writeln(i)
    }
    
    for (var i:int64 = 0; i < 1):
    {
        writeln(i)
    }
    
    for (var i:float64 = 0; i < 1; i += .03):
    {
        writeln(i)
    }

    
    // foreach(variable, sequence)
    foreach(var primary:string, ["red", "yellow", "blue"]):
    {
        writeln(primary)
    }
}
\end{DoxyVerbInclude}

Dictionaries are a little trickier.

fun main():no_value
