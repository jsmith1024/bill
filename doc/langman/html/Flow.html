While


# whileloop.bill
# while loop syntax sample

fun main(argsv):int8
{
    while true:
    {
        # This is the loop that never ends.
    }
    
    while 1 > 3:
    {
        # This loop is skipped.
    }
    
//     var $limit:vector   = $argsv[1]
    var i:int8 = 0
    while i < argsv[1]:
    {
        writeln(i)
        i ++
    }
}

# for loop syntax sample

/*  @fn     main
 *  @brief  simple forloop
 */
fun main():no_value
{
    for var i:int64 = 0 to 10:
    {
        writeln(i)
    }
    
    for (var i:int64 = 0; i < 1):
    {
        writeln(i)
    }
    
    for (var i:float64 = 0; i < 1; i += .03):
    {
        writeln(i)
    }

    
    # foreach (sequence types only)

//     }
}
